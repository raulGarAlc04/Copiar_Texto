interface interface1 {
    fun metodo1() = println("Este es el primer metodo")
}

interface interface2 {
    fun metodo2() = println("Este es el segundo metodo")
}

class claseHerencia : interface1, interface2 {
    override fun metodo1() = println("He modificado el primer metodo")
    override fun metodo2() = println("He modificado el segundo metodo")
}
    
fun main() {
    val variablePrueba = claseHerencia()
    variablePrueba.metodo1()
    variablePrueba.metodo2()
}
