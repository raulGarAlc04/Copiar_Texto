open class Persona(var nombre: String, var edad: Int) {
    fun datosPersona() = println("Nombre: ${nombre}, Edad: ${edad}")
}

class Empleado(nombre: String, edad: Int, var sueldo: Float): Persona(nombre, edad) {
    fun pagaImpuestos() {
        if(sueldo > 2000) {
            println("El empleado ${nombre} debe pagar impuestos porque su sueldo es de ${sueldo}")
        } else {
            println("El empleado ${nombre} NO debe pagar impuestos porque su sueldo es de ${sueldo}")
        }
    }
}
    
fun main() {
	val emp = Empleado("Raul", 19, 2075.78f)
    emp.pagaImpuestos()
}
